yum install gcc zlib-devel bzip2 bzip2-devel readline readline-devel sqlite sqlite-devel openssl openssl-devel git

- gcc
- zlib-devel
- bzip2
- bzip2-devel
- readline
- readline-devel
- sqlite
- sqlite-devel
- openssl
- openssl-devel
- git
